% Создаём счётчики для нумерации
\newcounter{theorem}
\newcounter{lemma}[theorem]   % леммы нумеруются вместе с теоремами
\newcounter{definition}

% Окружение для теоремы
\newenvironment{theorem}[1][]{%
  \par\medskip\noindent
  \refstepcounter{theorem}%
  \textbf{Теорема \thetheorem.}%
  \ifx\relax#1\relax\else\ {\normalfont(#1)}\fi
  \par\nobreak\noindent\ignorespaces
}{%
  \par\medskip
}

% Окружение для леммы
\newenvironment{lemma}[1][]{%
  \par\medskip\noindent
  \refstepcounter{lemma}%
  \textbf{Лемма \thelemma.}%
  \ifx\relax#1\relax\else\ {\normalfont(#1)}\fi
  \par\nobreak\noindent\ignorespaces
}{%
  \par\medskip
}

% Окружение для определения
\newenvironment{definition}[1][]{%
  \par\medskip\noindent
  \refstepcounter{definition}%
  \textbf{Определение \thedefinition.}%
  \ifx\relax#1\relax\else\ {\normalfont(#1)}\fi
  \par\nobreak\noindent\ignorespaces
}{%
  \par\medskip
}

% Окружение для доказательства
\newenvironment{proof}{%
  \par\medskip\noindent
  \textbf{Доказательство. }%
  \par\nobreak\noindent\ignorespaces
}{%
  \par\nobreak
  \noindent\hfill$\square$%
  \par\medskip
}

% Окружение для замечания (без номера)
\newenvironment{note}{%
  \par\medskip\noindent
  \textbf{Замечание. }%
  \par\nobreak\noindent\ignorespaces
}{%
  \par\medskip
}
